rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function isOwnerEmail() {
      return signedIn() && request.auth.token.email == "legendofarrie@gmail.com";
    }

    match /user/{userId} {
      allow read: if signedIn();
      allow create: if isOwner(userId);
      allow update: if isOwner(userId) || isOwnerEmail();
      allow delete: if false;
    }

    match /user_settings/{userId} {
      allow read: if signedIn();
      allow create, update: if isOwner(userId);
      allow delete: if false;
    }

    match /public_nicknames/{nicknameId} {
      allow get: if true;
      allow list: if false;
      allow create: if signedIn()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.nicknameLower == nicknameId
        && !exists(/databases/$(database)/documents/public_nicknames/$(nicknameId));
      allow update: if signedIn()
        && resource.data.userId == request.auth.uid
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.nicknameLower == nicknameId;
      allow delete: if signedIn() && resource.data.userId == request.auth.uid;
    }

    match /notifications/{notificationId} {
      allow read: if signedIn() && resource.data.toUserId == request.auth.uid;
      allow create: if signedIn()
        && request.resource.data.toUserId is string
        && (
          (request.resource.data.fromUserId == request.auth.uid) ||
          (!("fromUserId" in request.resource.data) && isOwnerEmail())
        );
      allow update: if signedIn()
        && resource.data.toUserId == request.auth.uid
        && request.resource.data.diff(resource.data).changedKeys().hasOnly(["read"]);
      allow delete: if signedIn()
        && (resource.data.toUserId == request.auth.uid || isOwnerEmail());
    }

    match /conversations/{conversationId} {
      allow read: if signedIn() && request.auth.uid in resource.data.members;
      allow create: if signedIn()
        && request.resource.data.members is list
        && request.auth.uid in request.resource.data.members;
      allow update: if signedIn() && request.auth.uid in resource.data.members;
      allow delete: if false;

      match /messages/{messageId} {
        allow read: if signedIn()
          && request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.members;
        allow create: if signedIn()
          && request.resource.data.senderId == request.auth.uid
          && request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.members;
        allow update: if signedIn()
          && request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.members
          && request.resource.data.diff(resource.data).changedKeys().hasOnly(["reactions"])
          && request.resource.data.reactions is map
          && (
            (!("reactions" in resource.data)
              && request.resource.data.reactions.keys().hasOnly([request.auth.uid]))
            || request.resource.data.reactions.diff(resource.data.reactions).changedKeys().hasOnly([request.auth.uid])
          );
        allow delete: if false;
      }
    }

    match /video_calls/{callId} {
      allow read: if signedIn()
        && (resource.data.callerId == request.auth.uid
          || resource.data.calleeId == request.auth.uid);
      allow create: if signedIn()
        && request.resource.data.callerId == request.auth.uid
        && request.resource.data.calleeId is string;
      allow update: if signedIn()
        && (resource.data.callerId == request.auth.uid
          || resource.data.calleeId == request.auth.uid);
      allow delete: if false;

      match /callerCandidates/{candidateId} {
        allow read: if signedIn()
          && (get(/databases/$(database)/documents/video_calls/$(callId)).data.callerId == request.auth.uid
            || get(/databases/$(database)/documents/video_calls/$(callId)).data.calleeId == request.auth.uid);
        allow create: if signedIn()
          && get(/databases/$(database)/documents/video_calls/$(callId)).data.callerId == request.auth.uid;
        allow update, delete: if false;
      }

      match /calleeCandidates/{candidateId} {
        allow read: if signedIn()
          && (get(/databases/$(database)/documents/video_calls/$(callId)).data.callerId == request.auth.uid
            || get(/databases/$(database)/documents/video_calls/$(callId)).data.calleeId == request.auth.uid);
        allow create: if signedIn()
          && get(/databases/$(database)/documents/video_calls/$(callId)).data.calleeId == request.auth.uid;
        allow update, delete: if false;
      }
    }

    match /live_rooms/{roomId} {
      function liveRoom() {
        return get(/databases/$(database)/documents/live_rooms/$(roomId)).data;
      }

      function isLiveHost() {
        return signedIn() && liveRoom().hostId == request.auth.uid;
      }

      allow read: if signedIn();
      allow create: if signedIn()
        && request.resource.data.hostId == request.auth.uid;
      allow update: if signedIn()
        && (
          isLiveHost()
          || request.resource.data.diff(resource.data).changedKeys().hasOnly(["viewerCount"])
        );
      allow delete: if false;

      match /messages/{messageId} {
        allow read: if signedIn();
        allow create: if signedIn()
          && (
            request.resource.data.senderId == request.auth.uid
            || (request.resource.data.senderId == "system" && isLiveHost())
          );
        allow update: if false;
        allow delete: if signedIn() && isLiveHost();
      }

      match /viewers/{viewerId} {
        allow read: if signedIn();
        allow create, update, delete: if signedIn() && viewerId == request.auth.uid;
      }

      match /guests/{guestId} {
        allow read: if signedIn();
        allow create, delete: if signedIn()
          && (guestId == request.auth.uid || isLiveHost());
        allow update: if false;
      }

      match /join_requests/{requestId} {
        allow read: if signedIn()
          && (requestId == request.auth.uid || isLiveHost());
        allow create: if signedIn() && requestId == request.auth.uid;
        allow update, delete: if signedIn()
          && (isLiveHost() || requestId == request.auth.uid);
      }

      match /connections/{connectionId} {
        allow read: if signedIn();
        allow create, update: if signedIn()
          && (
            request.resource.data.offererId == request.auth.uid
            || request.resource.data.answererId == request.auth.uid
            || resource.data.offererId == request.auth.uid
            || resource.data.answererId == request.auth.uid
          );
        allow delete: if false;

        match /offerCandidates/{candidateId} {
          allow read: if signedIn();
          allow create: if signedIn()
            && get(/databases/$(database)/documents/live_rooms/$(roomId)/connections/$(connectionId)).data.offererId == request.auth.uid;
          allow update, delete: if false;
        }

        match /answerCandidates/{candidateId} {
          allow read: if signedIn();
          allow create: if signedIn()
            && get(/databases/$(database)/documents/live_rooms/$(roomId)/connections/$(connectionId)).data.answererId == request.auth.uid;
          allow update, delete: if false;
        }
      }

      match /muted/{userId} {
        allow read: if signedIn() && userId == request.auth.uid;
        allow create, delete: if signedIn() && isLiveHost();
        allow update: if false;
      }
    }

    match /blocks/{blockId} {
      allow read: if signedIn()
        && (resource.data.fromUserId == request.auth.uid
          || resource.data.toUserId == request.auth.uid);
      allow create: if signedIn()
        && request.resource.data.fromUserId == request.auth.uid;
      allow update: if false;
      allow delete: if signedIn()
        && resource.data.fromUserId == request.auth.uid;
    }

    match /likes/{likeId} {
      allow read: if signedIn()
        && (resource.data.fromUserId == request.auth.uid
          || resource.data.toUserId == request.auth.uid);
      allow create, update: if signedIn()
        && request.resource.data.fromUserId == request.auth.uid;
      allow delete: if signedIn() && resource.data.fromUserId == request.auth.uid;
    }

    match /dislikes/{dislikeId} {
      allow read: if signedIn()
        && (resource.data.fromUserId == request.auth.uid
          || resource.data.toUserId == request.auth.uid);
      allow create, update: if signedIn()
        && request.resource.data.fromUserId == request.auth.uid;
      allow delete: if signedIn() && resource.data.fromUserId == request.auth.uid;
    }

    match /weekend_plans/{planId} {
      allow read: if signedIn();
      allow create: if signedIn() && request.resource.data.creatorId == request.auth.uid;
      allow update: if signedIn()
        && (
          resource.data.creatorId == request.auth.uid
          || request.resource.data.diff(resource.data).changedKeys().hasOnly(["rsvpCount"])
        );
      allow delete: if signedIn() && resource.data.creatorId == request.auth.uid;

      match /rsvps/{rsvpId} {
        allow read: if signedIn()
          && (
            request.auth.uid == rsvpId
            || request.auth.uid == get(/databases/$(database)/documents/weekend_plans/$(planId)).data.creatorId
          );
        allow create: if signedIn() && request.auth.uid == rsvpId;
        allow update, delete: if false;
      }
    }

    match /payment_requests/{requestId} {
      allow read: if signedIn()
        && (resource.data.userId == request.auth.uid || isOwnerEmail());
      allow create: if signedIn() && request.resource.data.userId == request.auth.uid;
      allow update: if signedIn() && isOwnerEmail();
      allow delete: if false;
    }

    match /escort_listings/{listingId} {
      allow read: if signedIn();
      allow create: if signedIn()
        && listingId == request.auth.uid
        && request.resource.data.ownerId == request.auth.uid;
      allow update: if signedIn()
        && listingId == request.auth.uid
        && resource.data.ownerId == request.auth.uid;
      allow delete: if signedIn() && listingId == request.auth.uid;
    }
  }
}
